-- Table: public."Match"

-- DROP TABLE public."Match";

CREATE TABLE public."Match"
(
    "IDMatch" integer NOT NULL,
    "Date" timestamp with time zone NOT NULL,
    "Name" character varying(20) COLLATE pg_catalog."default" NOT NULL,
    "MatchState" "char" NOT NULL,
    "Sentence" character varying(60) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT "Match_pkey" PRIMARY KEY ("IDMatch"),
    CONSTRAINT "Sentence" FOREIGN KEY ("Sentence")
        REFERENCES public."Sentence" ("Sentence") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID
)

TABLESPACE pg_default;

ALTER TABLE public."Match"
    OWNER to postgres;

CREATE TABLE public."MatchHistory"
(
    "IDMatch" integer NOT NULL,
    "Move" character varying(10) COLLATE pg_catalog."default" NOT NULL,
    "PointsEarned" smallint NOT NULL,
    "LostPoints" smallint NOT NULL,
    "MancheNumber" integer NOT NULL,
    CONSTRAINT "MatchHistory_pkey" PRIMARY KEY ("IDMatch"),
    CONSTRAINT "IDMatch" FOREIGN KEY ("IDMatch")
        REFERENCES public."Match" ("IDMatch") MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE CASCADE
)

TABLESPACE pg_default;

ALTER TABLE public."MatchHistory"
    OWNER to postgres;

CREATE TABLE public."Participate"
(
    "Email" character varying(30) COLLATE pg_catalog."default" NOT NULL,
    "IDMatch" integer NOT NULL,
    "UserType" "char" NOT NULL,
    CONSTRAINT "Participate_pkey" PRIMARY KEY ("Email", "IDMatch"),
    CONSTRAINT "Email" FOREIGN KEY ("Email")
        REFERENCES public."User" ("Email") MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT "IDMatch" FOREIGN KEY ("IDMatch")
        REFERENCES public."Match" ("IDMatch") MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE NO ACTION
        NOT VALID
)

TABLESPACE pg_default;

ALTER TABLE public."Participate"
    OWNER to postgres;
